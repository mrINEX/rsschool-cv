binary search (C):


    #include <stdio.h>
    #include <cs50.h>
    
    int binarySearch(int key, int array[], int min, int max) {
      int middle;
      if(max<min)
        return -1;
      else
        middle = (min+max)/2;
        
      if(array[middle]<key)
        return binarySearch(key, array, middle+1, max);
      else if(array[middle]>key)
        return binarySearch(key, array, min, middle-1);
      else
        return middle;
    }
